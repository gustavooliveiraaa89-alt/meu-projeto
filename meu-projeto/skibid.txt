class Usuario:
    """Representa um usuário da biblioteca."""
    def __init__(self, nome, id_usuario):
        self.nome = nome
        self.id_usuario = id_usuario
        self.livros_emprestados = []

    def adicionar_livro_emprestado(self, livro):
        self.livros_emprestados.append(livro)

    def remover_livro_emprestado(self, livro):
        if livro in self.livros_emprestados:
            self.livros_emprestados.remove(livro)

    def __str__(self):
        """Representação em string do usuário."""
        return f"Usuário: {self.nome} (ID: {self.id_usuario})"

class Biblioteca:
    """Gerencia livros e usuários."""
    def __init__(self):
        self.catalogo_livros = {} # {isbn: Livro}
        self.registro_usuarios = {} # {id_usuario: Usuario}

    def adicionar_livro(self, livro):
        """Adiciona um novo livro ao catálogo."""
        if livro.isbn not in self.catalogo_livros:
            self.catalogo_livros[livro.isbn] = livro
            print(f"Livro '{livro.titulo}' adicionado.")
        else:
            print(f"Erro: Livro com ISBN {livro.isbn} já existe.")

    def adicionar_usuario(self, usuario):
        """Adiciona um novo usuário ao registro."""
        if usuario.id_usuario not in self.registro_usuarios:
            self.registro_usuarios[usuario.id_usuario] = usuario
            print(f"Usuário '{usuario.nome}' registrado.")
        else:
            print(f"Erro: Usuário com ID {usuario.id_usuario} já existe.")
